CREATE OR REPLACE PROCEDURE NDF (n IN NUMBER)
AS
  vname   employees.emp_name%TYPE;
  vsalary employees.salary%TYPE;
BEGIN
  SELECT emp_name, salary
  INTO vname, vsalary
  FROM employees
  WHERE emp_id = n;

  DBMS_OUTPUT.PUT_LINE('Name: ' || vname || ', Salary: ' || vsalary);

EXCEPTION
  WHEN NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE('No employee found with given ID');
  WHEN TOO_MANY_ROWS THEN
    DBMS_OUTPUT.PUT_LINE('Multiple employees found with given ID');
END;
/
Output: Procedure created.



CREATE OR REPLACE PROCEDURE ndf_emp (n IN NUMBER)
AS
  vname    employees.emp_name%TYPE;
  vsalary  employees.salary%TYPE;

   max_rows_found EXCEPTION;

  v_count NUMBER;
BEGIN
  
  SELECT COUNT(*)
  INTO v_count
  FROM employees
  WHERE emp_id = n;

  IF v_count > 1 THEN
    RAISE max_rows_found;

  ELSIF v_count = 1 THEN
    SELECT emp_name, salary
    INTO vname, vsalary
    FROM employees
    WHERE emp_id = n;

    DBMS_OUTPUT.PUT_LINE('Name: ' || vname || ', Salary: ' || vsalary);

  ELSE
    DBMS_OUTPUT.PUT_LINE('No employee found with EMP_ID ' || n);
  END IF;

EXCEPTION
  WHEN max_rows_found THEN
    DBMS_OUTPUT.PUT_LINE('Multiple employees found with EMP_ID ' || n);
END;
/
Output: Procedure created.



CREATE OR REPLACE PROCEDURE UDF (m1 IN NUMBER, m2 IN NUMBER, m3 IN NUMBER)
AS
    TOTAL NUMBER;
    PERCENTAGE NUMBER;
    Invalid_Marks EXCEPTION;
BEGIN
    IF m1 < 0 OR m1 > 100 OR m2 < 0 OR m2 > 100 OR m3 < 0 OR m3 > 100 THEN
        RAISE Invalid_Marks;
        
    ELSIF m1 > 50 AND m2 > 50 AND m3 > 50 THEN
        DBMS_OUTPUT.PUT_LINE('RESULT = PASS');
        
    ELSE
        DBMS_OUTPUT.PUT_LINE('RESULT = FAIL');
    END IF;

EXCEPTION
    WHEN Invalid_Marks THEN 
        DBMS_OUTPUT.PUT_LINE('Enter valid marks for m1, m2, m3 between 0 - 100');
END;
/
Output: Procedure created.
