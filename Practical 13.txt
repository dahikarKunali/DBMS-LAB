CREATE OR REPLACE TRIGGER trig_upt
after update on st_info
BEGIN
DBMS_OUTPUT.PUT_LINE('this is statement level trigger fired');
END;
/ 
Output: Trigger created.

update st_info set SUB1 = SUB1 - 10;
Output: 10 rows updated.


CREATE OR REPLACE TRIGGER trig_upt
after update on st_info for each Row
BEGIN
DBMS_OUTPUT.PUT_LINE('this is Row level trigger fired');
END;
/ 
Output: Trigger created.

update st_info set SUB1 = SUB1 + 10;
Output: 10 rows updated

DROP TRIGGER trig_upt;
Output: Trigger dropped.

update st_info set SUB1 = SUB1 - 10 where sub1 > 80;
Output: 2 rows updated.

CREATE OR REPLACE TRIGGER C_Trig_ins
BEFORE INSERT ON ST_INFO
FOR EACH ROW
BEGIN

        RAISE_APPLICATION_ERROR(-20001, 'Marks not in range');

END;
