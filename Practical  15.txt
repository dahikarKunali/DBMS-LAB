create or replace procedure max_min_salary
as
  v_max   emp2.salary%type;
  v_min   emp2.salary%type;
  v_upper emp2.salary%type;
  v_lower emp2.salary%type;
begin
  select max(salary) into v_max from emp2;
  select min(salary) into v_min from emp2;

  v_upper := v_min * 10;
  v_lower := v_max / 10;

  dbms_output.put_line('maximum salary: ' || v_max);
  dbms_output.put_line('minimum salary: ' || v_min);
  dbms_output.put_line('upper value: ' || v_upper);
  dbms_output.put_line('lower value: ' || v_lower);
end;
/
Output: Procedure created.

create or replace trigger emp2_salary_range
before insert or update on emp2
for each row
declare
  v_max   emp2.salary%type;
  v_min   emp2.salary%type;
  v_upper emp2.salary%type;
  v_lower emp2.salary%type;
begin
  select max(salary), min(salary) into v_max, v_min from emp2;

  v_upper := v_min * 10;
  v_lower := v_max / 10;

  if :new.salary < v_lower or :new.salary > v_upper then
    raise_application_error(-20010, 
      'Salary must be between ' || v_lower || ' and ' || v_upper);
  end if;
end;
/
Output: Trigger created.
